    Block Button

    • Code

                                                            <button type="button" class="btn btn-primary">Block Button</button>
                                                            <button type="button" class="btn btn-sm btn-info">Block Button</button>
                                                            <button type="button" class="btn btn-xs btn-success">Block Button</button>
                                                        </div>
                                                        <!-- end d-grid -->
